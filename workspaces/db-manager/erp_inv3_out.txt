CREATE VIEW [mirae].[MCA210TV]
			(PO_DT,		ITEM_CD,	CUST_CD,	PO_PRICE,	FOR_PRICE) AS

SELECT		DISTINCT	A.PO_DT,	B.ITEM_CD,	MAX(A.CUST_CD),	MAX(B.PO_PRICE),	MAX(B.FOR_PRICE)
FROM 		MCA200T A,	MCA210T B,
			(SELECT	D.ITEM_CD,	MAX(PO_DT) AS PO_DT FROM MCA200T C,MCA210T D
			 WHERE C.PO_NO = D.PO_NO AND C.PO_DT <= CONVERT(VARCHAR(8), GETDATE(), 112)
			 GROUP BY D.ITEM_CD) C
WHERE 		A.PO_NO		=	B.PO_NO
AND      	A.PO_DT 	= 	C.PO_DT
AND      	B.ITEM_CD 	= 	C.ITEM_CD
GROUP BY	B.ITEM_CD, A.PO_DT
