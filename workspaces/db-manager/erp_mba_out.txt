=== SP_MCA207_SEL1 프로시저 정의 ===

CREATE PROCEDURE SP_MCA207_SEL1
(
	@S_PJT_NO 	VARCHAR(20))
AS
BEGIN
	CREATE TABLE #TEMP
	(
		ITEM_KIND	VARCHAR(2),
		STS			VARCHAR(8),
		KIND_PO_AMT	NUMERIC(18, 4)
	)

	DECLARE @S_AMT NUMERIC(18, 4)

	INSERT	INTO #TEMP (ITEM_KIND, STS, KIND_PO_AMT)
	SELECT	B.ITEM_KIND,'일반발주' AS STS, SUM(A.PO_AMT) AS KIND_PO_AMT
      FROM  MCA210T A
			INNER JOIN BBA010T B
			ON A.ITEM_CD = B.ITEM_CD
     WHERE	A.PJT_NO = @S_PJT_NO
       AND  A.STS IN ('50','55','60','80','90')
     GROUP BY B.ITEM_KIND
	UNION ALL
	SELECT	C.ITEM_KIND,'재고발주' AS STS, SUM(ISNULL((A.ADV_QTY * B.AVG_PRICE), 0)) 
	  FROM	MAD111T A, IBA200T B, BBA010T C
	 WHERE	A.PJT_NO = @S_PJT_NO
	   AND	A.ITEM_CD = B.ITEM_CD
       AND	A.WH_CD = B.WH_CD 
	   AND	LEFT(A.ADV_DT,6) = B.BASE_YYYYMM
       AND	A.ITEM_CD = C.ITEM_CD
       AND  A.ADV_KIND = 'W'
	GROUP BY C.ITEM_KIND

	SELECT @S_AMT = SUM(KIND_PO_AMT) 
	  FROM #TEMP;

	SELECT ITEM_KIND, STS, KIND_PO_AMT, ROUND((KIND_PO_AMT / @S_AMT * 100),2) AS RATE
	  FROM #TEMP
END
